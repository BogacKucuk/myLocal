public static final String FIND_BY_CUSTOMER_NUMBER = """
      SELECT * FROM T_MONEY_WITHDRAWAL_DEPOSIT 
      WHERE PROCESS_DATE >= ADD_MONTHS(TRUNC(SYSDATE), ?) 
      AND CUSTOMER_NUMBER = ? 
      AND (
          (FINANCIAL_FAMILY_CODE = 'PC' AND FINANCIAL_SUB_FAMILY_CODE = 'KAS-KEN')
          OR
          (FINANCIAL_FAMILY_CODE = 'PR' AND FINANCIAL_SUB_FAMILY_CODE = 'HES-KAS-KEN')
      )
      ORDER BY PROCESS_DATE DESC 
      FETCH FIRST ? ROWS ONLY
      """;
