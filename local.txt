@Test
void whenFindByRentalSafeBoxIdAndStatus_shouldCalculateDebtCorrectly() {
    // Arrange
    List<DebtEntity> debts = new ArrayList<>();
    DebtEntity debtEntity1 = new DebtEntity();
    debtEntity1.setDebtAmount(new BigDecimal("150"));
    DebtEntity debtEntity2 = new DebtEntity();
    debtEntity2.setDebtAmount(new BigDecimal("50"));
    debts.add(debtEntity1);
    debts.add(debtEntity2);

    // Mock repository cevabı
    when(debtRepository.findByRentalSafeBoxIdAndStatus(Mockito.anyLong(), Mockito.anyString()))
            .thenReturn(debts);

    // Act
    BigDecimal debt = safeDepositBoxDebtService.findByRentalSafeBoxDebt(123L);

    // Assert
    // 150 + 50 = 200 olmalı
    assertThat(debt).isNotNull();
    assertThat(debt).isEqualTo(new BigDecimal("200"));
}
