// Yeni versiyon
const generateCheckboxStates = (response) => {
  // İçerisine ekleyeceğimiz tüm checkboxları tutacak dizi
  const checks = [];

  // Ortak bir fonksiyon yazarak recursive şekilde çocukları da ekleyelim
  const addDocsToChecks = (docs) => {
    docs.forEach((doc) => {
      // Her doküman için bir checkbox kaydı
      checks.push({
        id: doc.key,
        checked: false,
      });
      // Children varsa onları da ekle (recursive)
      if (doc.children && doc.children.length > 0) {
        addDocsToChecks(doc.children);
      }
    });
  };

  // bbtDocuments varsa, checks dizisine ekle
  if (response?.bbtDocuments?.length) {
    addDocsToChecks(response.bbtDocuments);
  }

  // notExistsDocs varsa, checks dizisine ekle
  if (response?.notExistsDocs?.length) {
    addDocsToChecks(response.notExistsDocs);
  }

  return checks;
};
