SELECT
	*
FROM
	T_MONEY_WITHDRAWAL_DEPOSIT tmwd
WHERE
	tmwd.CUSTOMER_NUMBER = '49701970'
	AND tmwd.PROCESS_DATE >= ADD_MONTHS(TRUNC(SYSDATE), -12)
	AND (
		(tmwd.FINANCIAL_FAMILY_CODE = 'PC' AND tmwd.FINANCIAL_SUB_FAMILY_CODE = 'KAS-KEN')
		OR
		(tmwd.FINANCIAL_FAMILY_CODE = 'PR' AND tmwd.FINANCIAL_SUB_FAMILY_CODE = 'HES-KAS-KEN')
	)
ORDER BY
	PROCESS_DATE DESC
FETCH FIRST 10 ROWS ONLY;
