type Branches = {
  unitNum: number | string;
  addressInfo: object;
};

function extractUniqueBranches(transactions: Transaction[]): Branches[] {
  const uniqueUnits = new Set<number>();

  return transactions.reduce<Branches[]>((acc, trx) => {
    if (trx.ownerUnitNum != null && !uniqueUnits.has(trx.ownerUnitNum)) {
      uniqueUnits.add(trx.ownerUnitNum);

      acc.push({
        unitNum: trx.ownerUnitNum,
        addressInfo: {} // şimdilik boş, ihtiyaca göre doldurabilirsin
      });
    }
    return acc;
  }, []);
}
