import static org.junit.jupiter.api.Assertions.*;
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.extension.ExtendWith;
import org.springframework.test.context.junit.jupiter.SpringExtension;

@ExtendWith(SpringExtension.class)
class SmartControlDocumentsTest {

    @Test
    void getDocument_ShouldReturnDocumentName_WhenIdExists() {
        // Arrange
        String documentId = "4"; // VEKALETNAME

        // Act
        String documentName = SmartControlDocuments.getDocument(documentId);

        // Assert
        assertEquals("Vekaletname", documentName);
    }

    @Test
    void getDocument_ShouldReturnNotFoundMessage_WhenIdDoesNotExist() {
        // Arrange
        String documentId = "99999";

        // Act
        String documentName = SmartControlDocuments.getDocument(documentId);

        // Assert
        assertEquals("99999-Bulunamadı", documentName);
    }

    @Test
    void getDocument_ShouldReturnCorrectDocumentName_ForMultipleIds() {
        // Arrange
        String documentId = "51"; // IMZA

        // Act
        String documentName = SmartControlDocuments.getDocument(documentId);

        // Assert
        assertEquals("Ticari İmza veya İmza", documentName);
    }
}
