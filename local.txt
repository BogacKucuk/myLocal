package com.ykb.corebanking.safedepositbox.safedepositboxbe.config;

import javax.sql.DataSource;

import org.springframework.boot.autoconfigure.jdbc.DataSourceProperties;
import org.springframework.boot.context.properties.ConfigurationProperties;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.context.annotation.Primary;
import org.springframework.jdbc.datasource.SimpleDriverDataSource;

@Configuration
public class OracleDataSourceConfig {

    /**
     * Okuyacağımız property prefix: spring.datasource
     * (yani application.yml’de spring.datasource.url, spring.datasource.username, vs.)
     */
    @Bean
    @Primary
    @ConfigurationProperties("spring.datasource")
    public DataSourceProperties oracleDataSourceProperties() {
        return new DataSourceProperties();
    }

    /**
     * Asıl Oracle datasource, JPA/Hibernate bu DS ile çalışacak.
     */
    @Bean
    @Primary
    public DataSource oracleDataSource() {
        DataSourceProperties props = oracleDataSourceProperties();
        return props.initializeDataSourceBuilder().type(SimpleDriverDataSource.class).build();
    }
}
