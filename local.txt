function formatTimestamp(timestamp: string): string {
  // Format kontrolü
  if (!/^\d{20}$/.test(timestamp)) {
    throw new Error('Timestamp sadece 20 adet rakamdan oluşmalıdır');
  }

  const year = timestamp.substring(0, 4);
  const month = timestamp.substring(4, 6);
  const day = timestamp.substring(6, 8);
  const hour = timestamp.substring(8, 10);
  const minute = timestamp.substring(10, 12);
  const second = timestamp.substring(12, 14);
  const millisecond = timestamp.substring(14, 17);
  const microsecond = timestamp.substring(17, 20);

  // Tarih geçerliliği kontrolü (opsiyonel)
  const dateObj = new Date(`${year}-${month}-${day}T${hour}:${minute}:${second}`);
  if (isNaN(dateObj.getTime())) {
    throw new Error('Geçersiz tarih formatı');
  }

  return `${year}-${month}-${day} ${hour}:${minute}:${second}.${millisecond}${microsecond}`;
}
