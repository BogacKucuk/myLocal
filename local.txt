const options = [
  { value: 'a', textLabel: 'Option 1' },
  { value: 'b', textLabel: 'Option 2' },
  { value: 'c', textLabel: 'Option 3' },
];
const defaultCheckedList = ['a'];

class CheckboxGroupExample extends React.Component {
  constructor(props) {
    super(props);
    this.state = {
      checkedList: defaultCheckedList,
      indeterminate: true,
      checkAll: 0,
    };
    this.onChange = this.onChange.bind(this);
    this.onCheckAllChange = this.onCheckAllChange.bind(this);
  }

  render() {
    return (
      <div>
        <div style={{ borderBottom: '1px solid #E9E9E9' }}>
          <Checkbox
            indeterminate={this.state.indeterminate}
            onChange={this.onCheckAllChange}
            checked={this.state.checkAll}
          >
            Check all
          </Checkbox>
        </div>
        <br />

        <CheckboxGroup
          onChange={this.onChange}
          value={this.state.checkedList}
          items={options}
        />
      </div>
    );
  }

  onChange(checkedList) {
    this.setState({
      checkedList,
      indeterminate:
        !!checkedList.length && checkedList.length < options.length,
      checkAll: checkedList.length === options.length,
    });
  }

  onCheckAllChange(e) {
    this.setState({
      checkedList: e.target.checked ? ['a', 'b', 'c'] : [],
      indeterminate: false,
      checkAll: e.target.checked,
    });
  }
}

<CheckboxGroupExample />;
