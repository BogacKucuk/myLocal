it('last trx anchor clicked on MainMoneyWithdrawComponent', async () => {
  console.log('test log');
  
  // Anchor'ı bul ve tıkla
  const anchor: EfaAnchorElement = await screen.findByTestId('last-transaction-link');
  expect(anchor).to.exist;
  
  fireEvent.click(anchor);
  
  // Biraz bekle - state güncellemesi için
  await new Promise(resolve => setTimeout(resolve, 300));
  
  // Popup'ı queryByTestId ile ara, sonra findByTestId ile doğrula
  let popup = screen.queryByTestId('last-transaction-popup');
  
  if (!popup) {
    // Popup hemen görünmüyorsa, waitFor ile tekrar dene
    await waitFor(() => {
      popup = screen.queryByTestId('last-transaction-popup');
      expect(popup).to.exist;
    }, { timeout: 5000 });
  }
  
  // Artık popup var, close button'ı bul
  const closeButton: EfaButtonElement = await screen.findByTestId('last-transactions-closeBtn');
  fireEvent.click(closeButton);
  
  // Popup'ın kapandığını doğrula
  await waitFor(() => {
    expect(screen.queryByTestId('last-transaction-popup')).to.not.exist;
  });
});
