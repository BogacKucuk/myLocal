// Örn: src/features/money-withdraw/MainMoneyWithdraw.tsx (veya dispatch'i nereye koyuyorsan)
import { useAppDispatch, useAppSelector } from '../../store';
import { transactionInfoActions } from '../../store/slices/transactionSlice';

// ...
const dispatch = useAppDispatch();

// Hesap sahibi müşteri numarası (primary) — her zaman string'e çek
const primaryCustomerNumberRaw = useAppSelector(
  (s) => s.customerInfoReducer.customerNumber as string | number | undefined
);
const primaryCustomerNumber = String(primaryCustomerNumberRaw ?? '');

// ... PDF başarıyla indirildikten sonra:
if (primaryCustomerNumber) {
  dispatch(
    transactionInfoActions.addExtraDocSet({
      customerNumber: primaryCustomerNumber,
      setCode: 'GISESET9',
    })
  );
}
