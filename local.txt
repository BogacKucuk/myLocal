ErrorComponentDto:
      type: object
      description: Error component information
      properties:
        severityCode:
          type: string
          description: Severity code
          example: "INFO"
        rollbackIndicator:
          type: string
          description: Rollback indicator
          example: "N"
        originServId:
          type: string
          description: Origin service identifier
          example: "SRV001"
        contextString:
          $ref: '#/components/schemas/ContextStringDto'
        messageTxt:
          type: string
          description: Message text
          example: "Operation completed successfully"
        returnCode:
          type: integer
          format: int32
          description: Return code
          example: 0
        reasonCode:
          type: integer
          format: int32
          description: Reason code
          example: 0
        checksum:
          type: string
          description: Checksum value
          example: "ABC123"
      example: {"severityCode": "INFO", "rollbackIndicator": "N", "originServId": "SRV001", "messageTxt": "Operation completed successfully", "returnCode": 0}
